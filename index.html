function transformPDFCoordinates(coords, viewport) { // Transformation des coordonnées PDF vers coordonnées DOM // Selon le format des coordonnées reçues de votre API OCR // Si les coordonnées sont déjà en pixels (relatifs au canvas rendu) if (coords.x < viewport.width && coords.y < viewport.height) { return { x: coords.x, y: coords.y, width: coords.width, height: coords.height }; } // Si les coordonnées sont en points PDF (origine en bas à gauche) // Transformation classique PDF -> DOM const transform = viewport.transform; const [a, b, c, d, e, f] = transform; // Conversion des coordonnées PDF vers les coordonnées du canvas const x = a * coords.x + c * coords.y + e; const y = b * coords.x + d * coords.y + f; // Correction pour l'inversion Y (PDF = bas-gauche, DOM = haut-gauche) const correctedY = viewport.height - y - coords.height; return { x: x, y: correctedY, width: coords.width, height: coords.height }; }
